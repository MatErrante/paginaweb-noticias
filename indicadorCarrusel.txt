<div class="carousel">
  <div class="carousel-item">
    <h3></h3>
    <p></p>
  </div>
</div>

$(document).ready(function(){
  var indicators = []; // Array para almacenar los indicadores
  var currentIndex = 0; // Índice del indicador actual

  // Función para obtener los indicadores de la API
  function getIndicators() {
    $.ajax({
      url: 'https://api.example.com/indicators', // URL de la API
      dataType: 'json',
      success: function(data) {
        indicators = data; // Almacenar los indicadores en el array
        showIndicator(); // Mostrar el primer indicador
        setInterval(function() {
          currentIndex++; // Incrementar el índice del indicador actual
          if (currentIndex == indicators.length) {
            currentIndex = 0; // Volver al primer indicador al llegar al final del array
          }
          showIndicator(); // Mostrar el siguiente indicador
        }, 5000); // Intervalo de tiempo entre cada transición (en milisegundos)
      }
    });
  }

  // Función para mostrar el indicador actual en el carrusel
  function showIndicator() {
    var indicator = indicators[currentIndex];
    $('.carousel-item h3').text(indicator.name); // Mostrar el nombre del indicador
    $('.carousel-item p').text(indicator.description); // Mostrar la descripción del indicador
  }

  // Obtener los indicadores al cargar la página
  getIndicators();

  // Inicializar el carrusel
  $('.carousel').slick({
    slidesToShow: 1, // Número de elementos que se muestran a la vez
    arrows: true, // Mostrar flechas de navegación
    dots: false, // Ocultar puntos de navegación
    infinite: true, // Repetir el carrusel desde el principio al llegar al final
    responsive: [
      {
        breakpoint: 768, // Breakpoint para ajustar el número de elementos que se muestran en diferentes tamaños de pantalla
        settings: {
          slidesToShow: 1
        }
      },
      {
        breakpoint: 480,
        settings: {
          slidesToShow: 1
        }
      }
    ]
  });
});
